// Programa de ejemplo en lenguaje decaf
class Program
{
    int temp;
   void swap(int xp, int yp)
   {
        temp = 1;
        xp = yp;
        yp = temp;
   }

   void SelectionSort(int arr, int n){
        int i, j, min_idx;

        for (i=0; i<n-1; i=i+1){
            min_idx = i;
            for (j=i+1; j<n; j=j+1){
                if (arr[j] <  arr[min_idx]){
                    min_idx = j;
                }
            }
            swap(arr[min_idx], arr[i]);
        }
   }
}
